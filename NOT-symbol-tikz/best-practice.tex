  \mathrel{\ooalign{$#1$\cr\hidewidth$#2$\hidewidth\cr}}%

  A \Longleftrightarrow B \\
  A \myrel{\Longleftrightarrow}{\slash} B \\
  A \leftrightsquigarrow B \\
  A \myrel{\leftrightsquigarrow}{\backslash} B \\
  A \myrel{\leftrightsquigarrow}{\tikz[scale=.1]{\duck[witch=black!50!gray, longhair=red!80!black, jacket=black!50!gray, magicwand]}} B
\end{gather*}
