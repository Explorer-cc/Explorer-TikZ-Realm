\documentclass{ctexart}
\usepackage{amsmath}
\usepackage{tikz}
%使用\DeclareRobustCommand封装一个\staircasedet命令
\DeclareRobustCommand\staircasedet{%
    \tikz[baseline = {(0,-.1)},scale=.25]{
         \draw (0,1) -- (1,1) -- (1,0) -- (2,0) -- (2,-1) -- (3,-1);
    }
}
\begin{document}
\[
    A = [\alpha_1, \alpha_2, \alpha_3, \cdots,\alpha_n] \overset{化为行简化阶梯型}{\longrightarrow}
    \Big[
        \tikz [baseline = {(0,-.1)},scale=.25] \draw (0,1) -- (1,1) -- (1,0) -- (2,0) -- (2,-1) -- (3,-1);
        \Big]
\]
\[
    B = [\beta_1, \beta_2, \beta_3, \cdots,\beta_n] \longrightarrow
    \Big[
        \staircasedet
        \Big]
\]
\end{document}
