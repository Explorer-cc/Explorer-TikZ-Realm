\documentclass[border=1mm, tikz]{standalone}
\tikzset{
    every picture/.style={line cap=round, line join=round},
    pics/mycart/.style={
        code={
            \filldraw[line width=2pt, color=purple, fill=brown!30] 
                (-2.58,4.28) -- (-2.30,2.93) -- (0.59,2.67) -- (1.11,4.39) -- cycle;
            \node at (-0.75,3.5) {#1};
            \draw[line width=2pt] (-2.40,2.16) -- (0.84,2.16) -- (0.59,2.67);
            \draw[line width=2pt] (1.11,4.39) -- (1.39,4.78) -- (1.72,4.78);
            \draw[line width=2.8pt] (-2.17,1.78) circle[radius=0.25cm]
                coordinate (-wheel left);
            \draw[line width=2.8pt] (0.43,1.78) circle[radius=0.25cm]
                coordinate (-wheel right);
            \coordinate (-north west) at (-3,5);
            \coordinate (-north east) at (2,5);
            \coordinate (-south west) at (-3,1.5);
            \coordinate (-south east) at (2,1.5);
        }
    }
}
\begin{document}
\tikz {
    \pic (A) {mycart=x};
    \pic (B) at (6,2) {mycart=y};
    \draw (A-north west) -- (B-south east);
    \draw[red] (A-wheel left) -- (A-wheel right);
}
\end{document}
