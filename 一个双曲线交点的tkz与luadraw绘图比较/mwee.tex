\documentclass{standalone}
\usepackage{luadraw}
\usepackage{fouriernc}
\begin{document}
\begin{luadraw}{name=implicit_HY}
    local g = graph:new{window={-5.5,5.5,-3,3},size={10,10}}
    g:Linecap("round")
    local F = function(x,y) return x^2/3-y^2-1 end
    curve = implicit(F,-4.5,4.5,-3,3,{50,50})
    g:Dpolyline(curve,false,"thick")
    -- g:Dimplicit(F,{draw_options="thick"})
    g:Daxes({Z(0,0),1,1},{arrows="-Stealth"})
    local P,Q = Z(-4,0),Z(0,1)
    LL = {P,Q-P} -- line
    g:Dline(LL,"thick")
    B,A = table.unpack(interDL(LL,curve))
    g:Dlabel(
        "B",B,{pos="N"},
        "P",P,{},
        "Q",Q,{pos="NE"},
        "A",A,{}
    )
    g:Show()
\end{luadraw}
\end{document}
