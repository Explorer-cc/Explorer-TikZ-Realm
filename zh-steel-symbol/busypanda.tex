\newcommand*\steelD{%
  \setlength{\IHeight}{\fontcharht\font`I}%
  \setlength{\stroke}{\BusyPanda{0.08}\IHeight}%
  \tikz[baseline=0pt,line width=\stroke]{%
    \draw (-0.125\IHeight,0) -- (-0.125\IHeight,\IHeight);
    \draw (0.125\IHeight,0) -- (0.125\IHeight,\IHeight);
    \draw (-0.4\IHeight,.5\stroke) -- (0.4\IHeight,.5\stroke);
    \draw (-0.4\IHeight,\IHeight-.5\stroke) -- (0.4\IHeight,\IHeight-.5\stroke);
    \draw (0,.5\IHeight) ellipse [x radius=0.32\IHeight, y radius=0.25*(\IHeight-\stroke)];
  }%
}%
